import { Steps, Code, LinkCard, Aside } from "@astrojs/starlight/components";

   <Aside>

   If you haven't already created a discord bot application, then check this [*page*](/setup/createabot/).

   </Aside>

<Steps>

1. Install Node.js if not already installed (v22 recommended)

   <LinkCard
     title="Node.js Installation Guide"
     description="Follow the Node.js official website steps to install Node.js"
     href="https://nodejs.org/en/download/package-manager"
     target="_blank"
   />

2. Download the latest <a href="https://github.com/ralphkb/sentinel-tickets/releases" target="_blank">release</a> or clone the repository:

   <Code code="git clone https://github.com/ralphkb/sentinel-tickets.git" lang="shellscript" />

3. Change to the project directory, for example: `cd sentinel-tickets` or the directory where you uploaded the release files.

4. Install the dependencies:

   <Code code="npm install" lang="shellscript" />

5. Rename `.env.example` to `.env` and `config.yml.example` to `config.yml`

6. Open `.env` file and fill it with your bot's token, guild ID and client ID

7. Open `config.yml` to configure the settings and messages to your liking, make sure to properly configure the ticket categories

8. Start the bot:
      <Code code="npm start" lang="shellscript" />
</Steps>
